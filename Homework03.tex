\problem [10pt] Prove that by instruction that $1^2$ + $2^2$ + ... + $n^2$ = $\dfrac{n \times (n+1) \times (2n+1)}{6}$\\
\textbf{Solution:}\\
\textbf{Basic step:}\\
for n = 3; P(n): $1^{2} + 2^{2} + 3^{2}$ = $\dfrac{3 \times (3+1) \times (2 \times 3 + 1)}{6}$ = 14\\
$\rightarrow P(n) is true  for n =3$\\
\textbf{Inductive step:}\\
\begin{enumerate}
%1
\item n = k; P(k)\\
$1^2 + 2^2 + ... + k^2 = \dfrac{k \times (k+1) \times (2k+1)}{6}$\\
%2
\item n = k+1; P(k+1)\\
$1^2 + 2^2 + 3^2 + k^2 + (k+1)^2 = \dfrac {(k+1) \times (k+2) \times (2k+3)}{6}$\\
$\leftrightarrow \dfrac{k \times (k+1) \times (2k+1)}{6} + (k+1)^2 = \dfrac{(k+1) \times (k+2) \times (2k+3)}{6}$\\
$\leftrightarrow k \times (k+1) \times (2k +1) + 6 \times (k+1)^{2} = (k+1) \times (k+2) \times (2k+3)$\\
$\leftrightarrow k \times (2k+1) + 6 \times (k+1) = (k+1) \times (2k+1)$\\ 
$\rightarrow P(k+1) is true$\\
$\rightarrow Q.E.D.$
 
\end{enumerate}
